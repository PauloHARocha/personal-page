Sum of Squares Whitin SSW¶

$\large SSW = \sum\limits_{k=1}^{K}\sum\limits_{i=1}^{N_k} ||x_i - E[X_k]||^2$

In [4]:
# SSW - validated
def ssw(X, labels):
    k_range = np.unique(labels)
    k_range.sort()

    means = [np.mean(X[labels == l], axis=0) for l in k_range]
    means = np.reshape(means, [k_range.__len__(), X.shape[1]])
    return np.sum([np.linalg.norm(X[labels == l] - mean, axis=0)**2 for l, mean in zip(k_range, means)])

Gap Statistics¶

$\large Gap_k = \frac{1}{N_{ref}} \sum\limits_{ref=0}^{N_{ref}} \log{(\;SSW_{k,ref}\;)} - \log{(\;SSW_k\;)}$

$\large S_k = SD_k \sqrt{1 + \frac{1}{N_{ref}}} $

$ \large optimal_k = \textrm{smallest k such that } Gap(k) \geq Gap(k+1) - S_{k+1}$

def random_reference(X):
    randomRef = np.empty(shape=X.shape)
    for feat in range(X.shape[1]):
        randomRef[:, feat] = np.random.uniform(
            X[:, feat].min(), X[:, feat].max(), X[:, feat].shape)
    return randomRef

def random_ssw(X, k):
    
    randomRef = random_reference(X)
    
    model = KMeans(n_clusters=k, n_init=1).fit(X=randomRef)
   
    labels_ref = model.labels_ 
    
    return ssw(randomRef, labels_ref)


def gap(X, labels, nrefs=30):
    k = np.unique(labels).shape[0]
    refDisps = np.zeros(nrefs)

    for i in range(nrefs):
        refDisps[i] = random_ssw(X, k)

    origDisp = ssw(X, labels)

    # Calculate gap statistic
    gap_mean = np.mean(np.log(refDisps)) - np.log(origDisp)

    # Calculate gap statistic std
    gap_std = np.sqrt(1 + (1/nrefs))*np.std(np.log(refDisps))

    expec = np.mean(np.log(refDisps))
    ob = np.log(origDisp)

    return ob, expec, gap_mean, gap_std

def optimal_k(gap_means, gap_stds, k_min):
    for ig in range(gap_means.shape[0] - 1):
        if gap_means[ig] >= (gap_means[ig + 1] - gap_stds[ig + 1]):
            return ig + k_min
    return -1

def clustering_gap_proc(X, k_range, n_init=10):
    results = pd.DataFrame()
    
    for k in k_range:
        model = KMeans(n_clusters=k, n_init=10)
        model.fit(X)
        ob, expec, gap_mean, gap_std = gap(X.values, model.labels_)
        results = pd.concat([results, pd.DataFrame({
            'k': [k],
            'observed': [ob],
            'expected': [expec],
            'gap_mean': [gap_mean],
            'gap_std': [gap_std],
            'ssw': [model.inertia_]
        })], ignore_index=True)
    return results

def plot_gap_results(X, gap_results):
    randomRef = random_reference(X.values)
    f, axs = plt.subplots(1, 3, figsize=(20, 6))

    axs[0].scatter(X.values[:,0], X.values[:,1])
    axs[0].set_ylabel('Feature 2', fontsize=18)
    axs[0].set_xlabel('Feature 1', fontsize=18)
    axs[0].set_title('Data', fontsize=18)

    axs[1].plot(gap_results['k'], gap_results['ssw'], marker='o')
    axs[1].set_ylabel('SSW', fontsize=18)
    axs[1].set_xlabel('k', fontsize=18)
    axs[1].set_title('SSW', fontsize=18)

    axs[2].scatter(randomRef[:,0], randomRef[:,1])
    axs[2].set_ylabel('Uniform Distribution of Feature 2', fontsize=18)
    axs[2].set_xlabel('Uniform Distribution of Feature 1', fontsize=18)
    axs[2].set_title('Random Data', fontsize=18)


    f, axs = plt.subplots(1, 2, figsize=(14, 6))

    axs[0].plot(gap_results['k'], gap_results['observed'], marker='o', label='observed')
    axs[0].plot(gap_results['k'], gap_results['expected'], marker='s', label='expected')
    axs[0].set_ylabel('log(SSW)', fontsize=18)
    axs[0].set_xlabel('k', fontsize=18)

    axs[0].legend()

def plot_optimal_k(gap_results, k_min):
    k_op = optimal_k(gap_results['gap_mean'].values, gap_results['gap_std'], k_min=k_min)
    f, ax = plt.subplots(figsize=(14, 6))

    ax.errorbar(gap_results['k'], gap_results['gap_mean'], yerr=gap_results['gap_std'], label=None)
    ax.scatter(k_op, gap_results['gap_mean'].values[k_op - k_min],
               label=f'optimal k = {k_op}', facecolors='none', edgecolors='r', s=300)

    ax.set_ylabel('Gap', fontsize=18)
    ax.set_xlabel('k', fontsize=18)
    ax.set_title('Gap Statistic', fontsize=18)
    ax.legend();
